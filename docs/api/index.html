Authentication Examples

Use the Bearer token in the Authorization header:

# Get an access token first via OAuth 2.0
TOKEN=$(curl -s -X POST http://localhost:8006/oauth/token \
  -H "Content-Type: application/json" \
  -d '{"grant_type":"client_credentials","client_id":"your_client_id","client_secret":"your_client_secret"}' \
  | jq -r '.access_token')

# Use the token in API requests
curl -X GET http://localhost:8001/api/v1/identity/users \
  -H "Authorization: Bearer $TOKEN" \
  -H "Content-Type: application/json"

Using fetch API:

async function getAccessToken() {
  const response = await fetch('http://localhost:8006/oauth/token', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      grant_type: 'client_credentials',
      client_id: 'your_client_id',
      client_secret: 'your_client_secret'
    })
  });
  const data = await response.json();
  return data.access_token;
}

const token = await getAccessToken();

const response = await fetch('http://localhost:8001/api/v1/identity/users', {
  headers: {
    'Authorization': `Bearer ${token}`,
    'Content-Type': 'application/json'
  }
});
const users = await response.json();

Using requests library:

import requests

# Get access token
token_response = requests.post(
    'http://localhost:8006/oauth/token',
    json={
        'grant_type': 'client_credentials',
        'client_id': 'your_client_id',
        'client_secret': 'your_client_secret'
    }
)
access_token = token_response.json()['access_token']

# Use the token
headers = {
    'Authorization': f'Bearer {access_token}',
    'Content-Type': 'application/json'
}
response = requests.get(
    'http://localhost:8001/api/v1/identity/users',
    headers=headers
)
users = response.json()

Using http.Client:

package main

import (
    "context"
    "encoding/json"
    "net/http"
    "strings"
)

type TokenResponse struct {
    AccessToken string `json:"access_token"`
}

func getAccessToken() (string, error) {
    payload := `{"grant_type":"client_credentials",` +
        `"client_id":"your_client_id",` +
        `"client_secret":"your_client_secret"}`

    resp, err := http.Post(
        "http://localhost:8006/oauth/token",
        "application/json",
        strings.NewReader(payload),
    )
    if err != nil {
        return "", err
    }
    defer resp.Body.Close()

    var tokenResp TokenResponse
    if err := json.NewDecoder(resp.Body).Decode(&tokenResp); err != nil {
        return "", err
    }
    return tokenResp.AccessToken, nil
}

func main() {
    token, _ := getAccessToken()

    req, _ := http.NewRequest(
        "GET",
        "http://localhost:8001/api/v1/identity/users",
        nil,
    )
    req.Header.Set("Authorization", "Bearer "+token)
    req.Header.Set("Content-Type", "application/json")

    client := &http.Client{}
    resp, _ := client.Do(req)
    defer resp.Body.Close()
}
